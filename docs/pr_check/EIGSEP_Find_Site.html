def dbu_to_Tb(dbu, B=250e6/1026, R=377, FoV=2*np.pi, wavelen=C/100e6):
    """
    Convert electric field strength in dBu to antenna temperature in Kelvin.

    Parameters:
    dbu (float): Electric field strength in dBu.
    B (float): Bandwidth in Hz.
    R (float): Characteristic impedance of free space in ohms. Default 377
    FoV (float): Solid angle in steradians. Default 2pi
    wavelen (float): Wavelength [m]. Default for 100 MHz.

    Returns:
    float: Brightness temperature in Kelvin.
    """
    k_B = 1.38e-23  # Boltzmann's constant in J/K
    # Convert dBu to electric field strength in V/m
    e_field = 10**((dbu - 120) / 20)
    # Calculate the power received by the antenna in watts
    S = e_field**2 / R  # flux
    S_Jy = S / B * 1e26  # flux density, Jy
    T_b = S_Jy / 1e26 * wavelen**2 / (2 * k_B * FoV)
    #P = (e_field**2 * Aeff) / R

    # Calculate antenna temperature in Kelvin
    #T_a = P / (k_B * B)

    return T_b

class KML:
    def __init__(self, filename, namespace={'kml': 'http://www.opengis.net/kml/2.2'}):
        tree = ET.parse(filename)
        self.root = tree.getroot()
        self.namespace = namespace
        self.parse()
        #self._dbu = 60

    def parse(self):
        for placemark in self.root.findall('.//kml:Placemark', self.namespace):
            linestring = placemark.find('.//kml:LineString', self.namespace)
            if linestring is not None:
                coords_text = linestring.find('.//kml:coordinates', self.namespace).text.strip()
                coordinates = np.array([
                    tuple(map(float, coord.split(','))) for coord in coords_text.split()
                ])
                self.coords = coordinates[:, :2]
            else:
                point = placemark.find('.//kml:Point', self.namespace)
                if point is not None:
                    coords_text = point.find('.//kml:coordinates', self.namespace).text.strip()
                    self.lon, self.lat, _ = tuple(map(float, coords_text.split(',')))

class FCC_DataBase:
    def __init__(self, cache_file=CACHE_FILE, path=PATH,
                 lat_range=(33, 43), lon_range=LON_RANGE, fq_range_mhz=FQ_RANGE_MHZ):
        self._cache_file = cache_file
        self.APP_LOCATION_FILE = os.path.join(path, 'app_location.dat')
        self.FACILITY_FILE     = os.path.join(path, 'facility.dat')
        self.APPLICATION_FILE  = os.path.join(path, 'application.dat')
        self.lat_range = lat_range
        self.lon_range = lon_range
        self.fq_range_mhz = fq_range_mhz

    def clear_cache(self):
        if os.path.exists(self._cache_file):
            os.remove(self._cache_file)

    def read(self):
        if not os.path.exists(self._cache_file):
            _id, lat, lon = self._read_app_location(self.APP_LOCATION_FILE, self.lat_range, self.lon_range)
            id2callsign = {}
            _id0, _callsign0 = self._read_application(self.APPLICATION_FILE)
            id2callsign.update(dict(zip(_id0, _callsign0)))
            _id1a, _id1b, _callsign1, _fqs = self._read_facility(self.FACILITY_FILE, self.fq_range_mhz)
            # XXX do something with fqs
            id2callsign.update(dict(zip(_id1a, _callsign1)))
            id2callsign.update(dict(zip(_id1b, _callsign1)))
            id2freq = {}
            id2freq.update(dict(zip(_id1a, _fqs)))
            id2freq.update(dict(zip(_id1b, _fqs)))
            fqs = np.array([id2freq.get(i, 0) for i in _id])  # MHz to Hz
            np.savez(self._cache_file, ids=_id.astype('S32'), lats=lat, lons=lon, freqs=fqs)
        npz = np.load(self._cache_file)
        self.ids = npz['ids']
        self.lats = npz['lats']
        self.lons = npz['lons']
        self.freqs = npz['freqs']
        
    def _read_app_location(self, filename, lat_range, lon_range):
        csv = pandas.read_csv(filename, sep='|', low_memory=False)
        ids = csv['aloc_aapp_application_id']
        lat = csv['aloc_lat_deg'] + csv['aloc_lat_mm'] / 60 + csv['aloc_lat_ss'] / 3600
        lon = -(csv['aloc_long_deg'] + csv['aloc_long_mm'] / 60 + csv['aloc_long_ss'] / 3600)
        in_range = (lat >= lat_range[0]) & (lat <= lat_range[1]) & (lon >= lon_range[0]) & (lon <= lon_range[1])
        return ids[in_range], lat[in_range], lon[in_range]

    def _read_facility(self, filename, fq_range):
        csv = pandas.read_csv(filename, sep='|', low_memory=False)
        fq = csv['frequency']  # MHz
        #state = csv['community_served_state']
        select_fq = (fq >= fq_range[0]) & (fq <= fq_range[1])
        select = select_fq
        callsigns = csv['callsign'][select]
        return csv['latest_filing_version_id'][select], csv['license_filing_id'][select], csv['callsign'][select], fq[select] * 1e6

    def _read_application(self, filename):
        csv = pandas.read_csv(filename, sep='|', low_memory=False)
        return csv['aapp_application_id'], csv['aapp_callsign']

Digital Elevation Map (grayscale, m) of the western US with areas on public land with topographic features $> 100 m$ on either side in green, and the deepest $>95$ percentile in pink. FCC-registered FM and TV stations are shown in magenta and orange, respectively, with crosses at the transmitter and shading inside modeled 60 dBu contours. Blue circles indicate registered, but possibly inactive, FM and TV antennas. White circles indicate city centers. The yellow X marks the selected EIGSEP site.
